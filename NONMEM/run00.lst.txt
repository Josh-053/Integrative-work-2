$PROBLEM    CeftriaxPro Base Model Development
$INPUT      ID TIME DOSE=AMT RATE CONC=DV CONC_ID BW SEX ALB SCR eGFR
            AGE ETHNI PR DVID EVID
$DATA      dataset.csv IGNORE=@ IGNORE=(CONC_ID.EQ.1)
 ;; ignores rows showing total plasma concentration
$SUBROUTINE ADVAN1 ;; 1-compartment model
            TRANS2 ;; CL,V
$PK
TVCL = THETA(1)
  CL = TVCL * EXP(ETA(1))
 TVV = THETA(2)
   V = TVV * EXP(ETA(2))
   K = CL/V
  S2 = V/1                ;; scale prediction based on DOSE (mmol) and DV (mmol/L)

$THETA  (0,1.16,3.26) ; ; From Telles et al. (1),max range doubled
 (0,10.04,22.52) ; _; From Telles et al. (1),max range doubled
; explore dataset & literature, SAME order as PK
$OMEGA  0.1  ;         __  ; not yet known, outcome picked for next models
 0.1  ;        ___  ; not yet known, outcome picked for next models
 ;; variance covariance matrix for interindividual variability
$SIGMA  0.1  ;       ____  ; EPS(1) for proportional error, not yet known, outcome picked for next models
;; variance covariance matrix for residual error
$ERROR
IPRED=F
Y=IPRED*(1+EPS(1)) ;; proportional = EPS(1)
W = SQRT((IPRED*SQRT(SIGMA(1,1)))**2)
IRES  = CONC - IPRED
IWRES = IRES / W

$ESTIMATION METHOD=1 INTERACTION ;; FOCE-I
            MAXEVAL=9999 SIG=3 PRINT=5 NOABORT POSTHOC
 ;; second derivatives using -2log(likelihood),blank means sandwich method
$COVARIANCE
